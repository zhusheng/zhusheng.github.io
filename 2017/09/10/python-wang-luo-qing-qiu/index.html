python网络请求，介绍了urllib库、request库以及网络请求状态码。

urllib库网络请求

urllib是python内置的网络请求库。

# -*- coding: utf-8 -*-
from urllib import request, parse

def get():
    url = 'http://www.baidu.com'
    response = request.urlopen(url)

    # 打印二进制形式的响应内容
    #print(response.read())
    #  读取网页内容 将二进制内容按照utf-8格式解码 生成字符串格式
    print(type(response.read().decode('utf-8')))


def post():
    data = bytes(parse.urlencode({
        'word':'hello'
    }), encoding='utf-8')
    response = request.urlopen('http://httpbin.org/post', data = data)
    print(response.read())

def header():
    url = "http://httpbin.org/post"
    headers = {
        # 伪装一个火狐浏览器
        "User-Agent": 'Mozilla/4.0 (compatible; MSIE 5.5; Windows NT)',
        "host": 'httpbin.org'
    }
    dict = {
        "name": "Germey"
    }
    data = bytes(parse.urlencode(dict), encoding="utf8")
    # 利用更强大的 Request 类来构建一个请求
    req = request.Request(url=url, data=data, headers=headers, method="POST")

    response = request.urlopen(req)
    print(response.read().decode("utf-8"))

    # 也可以通过更加灵活的方式添加headers   如:
    # req.add_header('User-Agent', 'Mozilla/6.0 (iPhone; CPU iPhone OS 8_0 like Mac OS X) AppleWebKit/536.26 (KHTML, like Gecko) Version/8.0 Mobile/10A5376e Safari/8536.25')
    # req.add_header('Referer', 'https://passport.weibo.cn/signin/login?entry=mweibo&res=wel&wm=3349&r=http%3A%2F%2Fm.weibo.cn%2F')

if __name__ == '__main__':
    header()

requests库网络请求

requests是一个python 的第三方网络请求库，更加常用，通过该库可以很方便的实现get、post、put、delete、patch等请求方法。

参考链接：https://www.cnblogs.com/caesura-k/p/9638021.html

参考链接2：https://blog.csdn.net/byweiker/article/details/79234853

# -*- coding: utf-8 -*-
import requests
import json
from requests.exceptions import ReadTimeout
from requests.auth import HTTPBasicAuth


def get():
    """
    get请求
    :return:
    """
    response = requests.get('http://httpbin.org/get')
    print(response.text)

def get_with_params():
    """
    带参数的get请求
    :return:
    """
    response = requests.get("http://httpbin.org/get?name=germey&age=22")
    print(response.text)

def get_with_params2():
    """
    带参数的get请求
    :return:
    """
    data = {
        'name':'germey',
        'age':22
    }
    response = requests.get("http://httpbin.org/get", params=data)
    print(response.text)

def get_json():
    """
    将response以json的形式展示
    :return:
    """
    response = requests.get("http://httpbin.org/get")
    print(type(response.text))
    print(type(response.json()))
    print(response.json())
    print(json.loads(response.text))

def get_binary():
    """
    将response以二进制的形式展示
    :return:
    """
    response = requests.get("https://github.com/favicon.ico")
    print(type(response.text))
    print(type(response.content))
    print(response.content)

def add_headers():
    """
    添加请求头
    :return:
    """
    headers = {
        'User-Agent': 'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_11_4) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/52.0.2743.116 Safari/537.36'
    }
    response = requests.get("https://www.zhihu.com/explore", headers=headers)
    print(response.text)

def post():
    """
    post请求
    :return:
    """
    data = {'name': 'germey', 'age': '22'}
    response = requests.post("http://httpbin.org/post", data=data)
    print(response.text)

def get_response_attibutes():
    """
    获取响应对象的属性
    :return:
    """
    response = requests.get('http://www.jianshu.com')
    print(type(response.status_code), response.status_code)
    print(type(response.headers), response.headers)
    print(type(response.cookies), response.cookies)
    print(type(response.url), response.url)
    print(type(response.history), response.history)

def file_upload():
    """
    文件上传
    :return:
    """
    files = {'file': open('/Users/zhusheng/WorkSpace/Tmp/cookie.txt', 'rb')}
    response = requests.post("http://httpbin.org/post", files=files)
    print(response.text)

def get_cookies():
    """
    获取cookies
    :return:
    """
    response = requests.get("https://www.baidu.com")
    print(response.cookies)
    for key, value in response.cookies.items():
        print(key + '=' + value)

def maintain_cookies():
    """
    会话维持、模拟登陆
    如果某个响应中包含一些Cookie，你可以快速访问它们
    :return:
    """
    r = requests.get('http://www.google.com.hk/')
    print(r.cookies['NID'])
    print(tuple(r.cookies))

def send_cookies():
    """
    要想发送你的cookies到服务器，可以使用 cookies 参数：
    :return:
    """
    url = 'http://httpbin.org/cookies'
    cookies = {'testCookies_1': 'Hello_Python3', 'testCookies_2': 'Hello_Requests'}
    # 在Cookie Version 0中规定空格、方括号、圆括号、等于号、逗号、双引号、斜杠、问号、@，冒号，分号等特殊符号都不能作为Cookie的内容。
    r = requests.get(url, cookies=cookies)
    print(r.json())


def set_proxy():
    """
    设置代理
    :return:
    """
    proxies = {
        "http": "http://127.0.0.1:9743",
        "https": "https://127.0.0.1:9743",
    }

    response = requests.get("https://www.taobao.com", proxies=proxies)
    print(response.status_code)

def set_timeout():
    """
    设置超时
    :return:
    """
    try:
        response = requests.get("http://httpbin.org/get", timeout=0.5)
        print(response.status_code)
    except ReadTimeout:
        print('Timeout')

def set_auth():
    """
    设置认证
    :return:
    """
    response = requests.get("http://120.27.34.24:9001/", auth=HTTPBasicAuth("user", "123"))
    print(response.status_code)

def set_auth2():
    response = requests.get("http://120.27.34.24:9001/", auth=("user", "123"))
    print(response.status_code)


if __name__ == '__main__':
    set_proxy()
